                           {name: 'Baba', id: 101, marks : 23 },
                           {name: 'yaga', id: 200, marks : 45 },
                           {name: 'Wick', id: 115, marks : 75 } ]

Qu 7. This time we are required to print the total marks of the students with marks greater than 50 after a grace of 15 marks has been added to those students who scored less than 50 ?

    let studentRecords = [ {name: 'John', id: 123, marks : 98 },
                           {name: 'Baba', id: 101, marks : 23 },
                           {name: 'yaga', id: 200, marks : 45 },
                           {name: 'Wick', id: 115, marks : 75 } ]

    let sum=0,score;
    let marks = studentRecords.map(stu => {
      if (stu.marks > 50)
        sum +=stu.marks
      else
      {
        score = stu.marks+15
        if (score>50)
        {
          sum+=score
          score=0
        }
      }
      return sum
  });
  console.log("Sum of marks : ",sum);
